.back-to-top { position: fixed; display: none; width: 30px; text-align: center; right: 30px; bottom: 30px; z-index: 11; animation: action 1s infinite alternate; } $(window).scroll(function () { if ($(this).scrollTop() > 200) { $('.back-to-top').fadeIn('slow'); } else { $('.back-to-top').fadeOut('slow'); } }); $('.back-to-top').click(function () { $('html, body').animate({scrollTop: 0}, 1500, 'easeInOutExpo'); return false; });
